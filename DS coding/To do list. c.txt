#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Task {
    int id;
    char title[100];
    char dueDate[20];
    char priority[20];
    int completed;
    struct Task *next;
};

struct Task *head = NULL;
int nextId = 1;

void addTask(char title[], char dueDate[], char priority[]) {
    struct Task *newTask = (struct Task *)malloc(sizeof(struct Task));

    if (newTask == NULL) {
        printf("Memory allocation failed.\n");
        return;
    }

    newTask->id = nextId++;
    strcpy(newTask->title, title);
    strcpy(newTask->dueDate, dueDate);
    strcpy(newTask->priority, priority);
    newTask->completed = 0;
    newTask->next = NULL;

    if (head == NULL) {
        head = newTask;
    } else {
        struct Task *temp = head;
        while (temp->next != NULL) {
            temp = temp->next;
        }
        temp->next = newTask;
    }

    printf("Task added successfully.\n");
}

void displayTasks() {
    struct Task *temp = head;

    if (temp == NULL) {
        printf("\nNo tasks available.\n");
        return;
    }

    printf("\n========= TODO LIST =========\n");
    while (temp != NULL) {
        printf("ID: %d\n", temp->id);
        printf("Title: %s\n", temp->title);
        printf("Due Date: %s\n", temp->dueDate);
        printf("Priority: %s\n", temp->priority);
        printf("Status: %s\n", temp->completed ? "Completed" : "Pending");
        printf("-----------------------------\n");
        temp = temp->next;
    }
}

void deleteTask(int id) {
    struct Task *temp = head, *prev = NULL;

    if (temp != NULL && temp->id == id) {
        head = temp->next;
        free(temp);
        printf("Task deleted successfully.\n");
        return;
    }

    while (temp != NULL && temp->id != id) {
        prev = temp;
        temp = temp->next;
    }

    if (temp == NULL) {
        printf("Task not found.\n");
        return;
    }

    prev->next = temp->next;
    free(temp);
    printf("Task deleted successfully.\n");
}

void markCompleted(int id) {
    struct Task *temp = head;

    while (temp != NULL) {
        if (temp->id == id) {
            temp->completed = 1;
            printf("Task marked as completed.\n");
            return;
        }
        temp = temp->next;
    }

    printf("Task not found.\n");
}

void editTask(int id, char newTitle[], char newDate[], char newPriority[]) {
    struct Task *temp = head;

    while (temp != NULL) {
        if (temp->id == id) {
            strcpy(temp->title, newTitle);
            strcpy(temp->dueDate, newDate);
            strcpy(temp->priority, newPriority);
            printf("Task updated successfully.\n");
            return;
        }
        temp = temp->next;
    }

    printf("Task not found.\n");
}

void freeAllTasks() {
    struct Task *temp;
    while (head != NULL) {
        temp = head;
        head = head->next;
        free(temp);
    }
}

int main() {
    int choice, id;
    char title[100], dueDate[20], priority[20];

    while (1) {
        printf("\n====== ADVANCED TODO APP IN C ======\n");
        printf("1. Add Task\n");
        printf("2. Display Tasks\n");
        printf("3. Delete Task\n");
        printf("4. Mark Task Completed\n");
        printf("5. Edit Task\n");
        printf("6. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);
        getchar();

        switch (choice) {
            case 1:
                printf("Enter task title: ");
                fgets(title, sizeof(title), stdin);
                title[strcspn(title, "\n")] = '\0';

                printf("Enter due date (YYYY-MM-DD): ");
                fgets(dueDate, sizeof(dueDate), stdin);
                dueDate[strcspn(dueDate, "\n")] = '\0';

                printf("Enter priority (Low/Medium/High): ");
                fgets(priority, sizeof(priority), stdin);
                priority[strcspn(priority, "\n")] = '\0';

                addTask(title, dueDate, priority);
                break;

            case 2:
                displayTasks();
                break;

            case 3:
                printf("Enter task ID to delete: ");
                scanf("%d", &id);
                getchar();
                deleteTask(id);
                break;

            case 4:
                printf("Enter task ID to mark completed: ");
                scanf("%d", &id);
                getchar();
                markCompleted(id);
                break;

            case 5:
                printf("Enter task ID to edit: ");
                scanf("%d", &id);
                getchar();

                printf("Enter new title: ");
                fgets(title, sizeof(title), stdin);
                title[strcspn(title, "\n")] = '\0';

                printf("Enter new due date (YYYY-MM-DD): ");
                fgets(dueDate, sizeof(dueDate), stdin);
                dueDate[strcspn(dueDate, "\n")] = '\0';

                printf("Enter new priority (Low/Medium/High): ");
                fgets(priority, sizeof(priority), stdin);
                priority[strcspn(priority, "\n")] = '\0';

                editTask(id, title, dueDate, priority);
                break;

            case 6:
                freeAllTasks();
                printf("Exiting program...\n");
                return 0;

            default:
                printf("Invalid choice. Try again.\n");
        }
    }

    return 0;
}
